Posztok, amik a Telekom HU feed-jére kerülnek fel, bárki legyen is a posztoló
---------------------------------------------------------------------------------
select message from feed_all where fbi_object_reference='189246802995/feed'


-----------------------------------------------------------------------------------------------
Posztok, amik a Telekom HU feed-jére kerülnek fel, és maga a Telekom HU posztolta fel
-----------------------------------------------------------------------------------------------
select 
       message_id
      ,message 
      ,from_id
from feed_all
where
     from_id='189246802995'
and  fbi_object_reference='189246802995/feed' limit 100;


-----------------------------------------------------------------------------------------------
Kommentek, amelyek a Telekom HU feed-jére felkerült, Telekom HU által felrakott posztra érkeztek
-----------------------------------------------------------------------------------------------
select 
  distinct
       message_id
      ,message 
      ,from_id
      ,cmnt_message
from feed_all
where
     from_id='189246802995'
and  fbi_object_reference='189246802995/feed'
and  cmnt_message !='NULL';

-----------------------------------------------------------------------------------------------
Like-ok, amelyek a Telekom HU feed-jére felkerült, Telekom HU által felrakott posztra érkeztek
bárki is volt a lájkoló
-----------------------------------------------------------------------------------------------
select 
       distinct
       message_id
      ,message 
      ,liker_id
from feed_all
where
     from_id='189246802995'
and  fbi_object_reference='189246802995/feed'
and  liker_id !='NULL';


---------------------------------------------------------------------------------------------------------------
Like-ok száma kommentenként a Telekom HU által felrakott, Telekom Hu feedjére felrakott posztra érkeztek
bárki is volt a lájkoló
---------------------------------------------------------------------------------------------------------------
select message_id
       ,message
       ,count(*) like_osszesen
 from (
select 
       distinct
       message_id
      ,message 
      ,liker_id
from feed_all
where
    from_id='189246802995'
and fbi_object_reference='189246802995/feed'
and liker_id !='NULL')
group by message_id, message
