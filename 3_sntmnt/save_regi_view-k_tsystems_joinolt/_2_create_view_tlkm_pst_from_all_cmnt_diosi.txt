create or replace view sntmnt.tlkm_pst_from_all_cmnt as
select 
  distinct
       message_id
      ,message 
      ,from_id
      ,cmnt_message
      ,cmnt_id
      ,cmnt_sender_id
from mtsbi.feed_all
where
     fbi_object_reference='189246802995/feed'   -- telekom feed-jérõl van szó
and  cmnt_message !='NULL';                     -- van comment
